# Serial Plotter

```tsx
import SerialPlotter from '@hfr/serial-plotter/components';
import Webserial from '@tdev/webserial/component'
<Webserial
    baudRate={115200}
    hideLogs
    resetTrigger='::READY::'
    output={<SerialPlotter
        separator=' '
        config={{ timestamp: 0, ['Temperatur']: 1 }}
        minValue={24}
        timeFormatter='ms'
        grid={[5, 5]}
    />}
/>
```
